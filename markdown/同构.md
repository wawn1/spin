#### 同构

同构指的是react代码中有浏览器操作api, dom,事件等，需要在浏览器执行，只好服务器端执行一遍，浏览器再执行一遍react的js源码，绑定事件才能生效

####  静态资源解析

html引入js,js会在客户端执行

```js
// src/index.js
app.use(express.static("public"));

app.get("/", function (req, res) {
  res.send(`
  <html>
  <body>
  renderToString(<Home />)
  <script src='/index.js'></script>
  </body>
  </html>
  `);
});
// public/index.js
console.log("index.js");
```

#### 打包产生public的index.js

html引入了index.js, 只需要将需要客户端执行的代码，打包到public下的index.js就行了

```js
// webpack.client.js
const path = require("path");

module.exports = {
  entry: "./src/client/index.js",
  output: {
    filename: "index.js",
    path: path.resolve(__dirname, "public"),
  },
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        loader: "babel-loader",
        options: {
          presets: [
            "@babel/react",
            [
              "@babel/preset-env",
              {
                useBuiltIns: "usage",
                corejs: {
                  version: 3,
                },
                targets: {
                  chrome: "60",
                  firefox: "60",
                  ie: "9",
                  safari: "10",
                  edge: "17",
                },
              },
            ],
          ],
        },
      },
    ],
  },
  mode: "development",
};

```

```js
// src/index.js
import express from "express";
import Home from "./containers/Home";
import React from "react";
import {renderToString} from "react-dom/server";

const app = express();
const content = renderToString(<Home />);

app.use(express.static("public"));

app.get("/", function (req, res) {
  res.send(`
  <html>
  <body>
  <div id='root'>${content}</div>  // 客户端react代码执行需要
  <script src='/index.js'></script> // 打包好的客户端react代码
  </body>
  </html>
  `);
});

const server = app.listen(3000, function () {
  console.log(`server started at http://localhost:3000`);
});

```

const serverConfig = {
  target: "node",
  entry: "./src/server/index.js",
  output: {
    filename: "bundle.js",
    path: path.resolve(__dirname, "build"),
  },
